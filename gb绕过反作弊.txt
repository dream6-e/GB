local function loadAntiBanSystem()
    print("[Anti-Ban] 正在加载反作弊绕过系统...")
    
    -- ===== 基础配置 =====
    local ANTIBAN_DEBUG = false
    local Players = game:GetService("Players")
    local LocalPlayer = Players.LocalPlayer
    local RunService = game:GetService("RunService")
    
    -- ===== 执行器检测 =====
    local ExecutorCapabilities = {
        hasHookMetamethod = hookmetamethod ~= nil,
        hasHookFunction = hookfunction ~= nil,
        hasNewCClosure = newcclosure ~= nil,
        hasCheckCaller = checkcaller ~= nil,
        hasGetHui = gethui ~= nil,
        hasSynProtect = syn and syn.protect_gui ~= nil,
        hasGetNamecallMethod = getnamecallmethod ~= nil,
    }
    
    -- ===== GUI保护系统 =====
    local STEALTH_GUI_NAMES = {
        "RobloxGui", "CoreGui", "SystemGui", "PlayerListGui", "ChatWindow",
        "BackpackGui", "HealthGui", "TopBarApp", "SettingsHub", "InspectMenu",
        "RbxAnalyticsService", "RbxEventHandler", "RbxReplicatedFirst",
        "CoreScript_Internal", "PlayerDisplay_Handler", "SystemOverlay",
        "UIService_Manager", "CorePackages_Root", "VoiceChatShield",
    }
    
    local ProtectedGUIs = setmetatable({}, { __mode = "v" })
    local AlreadyProtectedGUIs = setmetatable({}, { __mode = "k" })
    local OurScriptGUIs = setmetatable({}, { __mode = "k" })
    
    local function getRandomStealthName()
        local base = STEALTH_GUI_NAMES[math.random(1, #STEALTH_GUI_NAMES)]
        return base .. "_" .. math.random(10000, 99999)
    end
    
    local function protectGUI(gui)
        if not gui or AlreadyProtectedGUIs[gui] then return end
        AlreadyProtectedGUIs[gui] = true
        OurScriptGUIs[gui] = true
        
        pcall(function()
            gui.Name = getRandomStealthName()
            
            if gethui then
                gui.Parent = gethui()
            elseif syn and syn.protect_gui then
                syn.protect_gui(gui)
                gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
            else
                gui.Parent = game:GetService("CoreGui")
            end
            
            pcall(function() gui.ResetOnSpawn = false end)
            pcall(function() gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling end)
            
            ProtectedGUIs[gui] = true
        end)
        
        if ANTIBAN_DEBUG then
            warn("[Anti-Ban] GUI保护完成: " .. gui.Name)
        end
    end
    
    -- ===== ESP保护系统 =====
    local ESPProtectedInstances = setmetatable({}, { __mode = "k" })
    
    local function protectESP(instance)
        if not instance or ESPProtectedInstances[instance] then return end
        pcall(function()
            ESPProtectedInstances[instance] = true
            for _, desc in ipairs(instance:GetDescendants()) do
                ESPProtectedInstances[desc] = true
            end
        end)
    end
    
    -- ===== 踢出/封禁拦截 =====
    local kickBlocked = 0
    local banBlocked = 0
    
    pcall(function()
        if hookfunction and newcclosure then
            hookfunction(LocalPlayer.Kick, newcclosure(function(_, reason)
                kickBlocked = kickBlocked + 1
                if ANTIBAN_DEBUG then
                    warn("[Anti-Ban] 拦截踢出 #" .. kickBlocked .. ": " .. tostring(reason))
                end
                return nil
            end))
        end
    end)
    
    -- ===== 远程调用拦截 =====
    local blockedRemotes = {
        "ReportPlayer", "BanPlayer", "KickPlayer", "LogExploit",
        "DetectedCheat", "SecurityViolation", "AnticheatReport", "FlagPlayer"
    }
    
    local suspiciousArgs = {
        "exploit", "cheat", "hack", "aimbot", "godmode", "inject", "executor"
    }
    
    pcall(function()
        if hookmetamethod and getnamecallmethod then
            hookmetamethod(game, "__namecall", function(self, ...)
                local method = getnamecallmethod()
                
                if method == "Kick" and self == LocalPlayer then
                    kickBlocked = kickBlocked + 1
                    return nil
                end
                
                if method == "FireServer" or method == "InvokeServer" then
                    local remoteName = ""
                    pcall(function() remoteName = self.Name end)
                    
                    for _, blocked in ipairs(blockedRemotes) do
                        if remoteName == blocked then
                            banBlocked = banBlocked + 1
                            if ANTIBAN_DEBUG then
                                warn("[Anti-Ban] 拦截远程: " .. remoteName)
                            end
                            return nil
                        end
                    end
                    
                    local args = {...}
                    for _, arg in pairs(args) do
                        if type(arg) == "string" then
                            local lowerArg = arg:lower()
                            for _, suspicious in ipairs(suspiciousArgs) do
                                if lowerArg:find(suspicious) then
                                    if ANTIBAN_DEBUG then
                                        warn("[Anti-Ban] 拦截可疑参数: " .. suspicious)
                                    end
                                    return nil
                                end
                            end
                        end
                    end
                end
                
                return nil
            end)
        end
    end)
    
    -- ===== 安全远程调用封装 =====
    _G.SafeFireServer = function(remote, ...)
        if remote and (remote:IsA("RemoteEvent") or remote:IsA("RemoteFunction")) then
            pcall(function()
                if remote:IsA("RemoteEvent") then
                    remote:FireServer(...)
                elseif remote:IsA("RemoteFunction") then
                    remote:InvokeServer(...)
                end
            end)
        end
    end
    
    _G.SafeGetAsync = function(url)
        pcall(function()
            if game.HttpGet then
                return game:HttpGet(url)
            else
                return game:GetService("HttpService"):GetAsync(url)
            end
        end)
        return ""
    end
    
    -- ===== 攻击限制系统 =====
    local KILLAURA_MAX_ATTACKS_PER_SECOND = 30
    local KILLAURA_MIN_DELAY = 0.03
    local KILLAURA_ATTACK_COUNTER = 0
    local KILLAURA_LAST_SECOND = os.time()
    
    local function canAttack()
        local currentSecond = os.time()
        if currentSecond ~= KILLAURA_LAST_SECOND then
            KILLAURA_LAST_SECOND = currentSecond
            KILLAURA_ATTACK_COUNTER = 0
        end
        return KILLAURA_ATTACK_COUNTER < KILLAURA_MAX_ATTACKS_PER_SECOND
    end
    
    local function recordAttack()
        KILLAURA_ATTACK_COUNTER = KILLAURA_ATTACK_COUNTER + 1
    end
    
    -- ===== 随机延迟系统 =====
    local function getRandomizedDelay(baseDelay)
        local variation = baseDelay * 0.2  -- 20% 随机变化
        return baseDelay + (math.random() * variation * 2 - variation)
    end
    
    local function humanizedWait(delay)
        local jitter = math.random() * 0.1 - 0.05  -- ±5% 随机抖动
        task.wait(delay + jitter)
    end
    
    -- ===== 弱引用表系统 =====
    local function createWeakTable()
        return setmetatable({}, { __mode = "k" })
    end
    
    local watchedTools = createWeakTable()
    local ESPData = createWeakTable()
    
    -- ===== 性能优化系统 =====
    local frameCounter = 0
    local UPDATE_INTERVAL = 5
    
    local function shouldUpdateThisFrame()
        frameCounter = frameCounter + 1
        if frameCounter >= UPDATE_INTERVAL then
            frameCounter = 0
            return true
        end
        return false
    end
    
    -- ===== 暴露全局函数 =====
    _G.protectGUI = protectGUI
    _G.protectESP = protectESP
    _G.getRandomStealthName = getRandomStealthName
    _G.canAttack = canAttack
    _G.recordAttack = recordAttack
    _G.getRandomizedDelay = getRandomizedDelay
    _G.humanizedWait = humanizedWait
    
    -- ===== 启动保护扫描 =====
    task.defer(function()
        while true do
            task.wait(15)
            pcall(function()
                for gui, _ in pairs(OurScriptGUIs) do
                    if gui and gui.Parent then
                        gui.Name = getRandomStealthName()
                    end
                end
            end)
        end
    end)
    
    print("[Anti-Ban] ✅ 反作弊系统加载完成")
    print("[Anti-Ban] 保护功能: GUI隐藏, 踢出拦截, 远程过滤, 攻击限制")
    
    return {
        protectGUI = protectGUI,
        protectESP = protectESP,
        SafeFireServer = _G.SafeFireServer,
        SafeGetAsync = _G.SafeGetAsync,
        canAttack = canAttack,
        recordAttack = recordAttack,
        getRandomizedDelay = getRandomizedDelay,
        humanizedWait = humanizedWait
    }
end

-- ===== 僵尸高亮功能 =====
local function processZombie(zombie)
    if not systemConfig.zombieHighlighter.Enabled then return end
    
    local function hasComponent(name)
        if zombie:FindFirstChild(name) then return true end
        for _, d in ipairs(zombie:GetDescendants()) do
            if d.Name == name then return true end
        end
        return false
    end
    
    local color
    local hasTorch = hasComponent("Torch")
    local hasCuirskull = hasComponent("Cuirskull")
    local hasEye = hasComponent("Eye")
    local hasAxe = hasComponent("Axe")
    
    if hasTorch then
        color = systemConfig.zombieHighlighter.Colors.Torch
    elseif hasCuirskull then
        color = systemConfig.zombieHighlighter.Colors.Cuirskull
    elseif hasEye and hasAxe then
        color = systemConfig.zombieHighlighter.Colors.Axe
    elseif hasEye then
        color = systemConfig.zombieHighlighter.Colors.Eye
    else
        color = systemConfig.zombieHighlighter.Colors.Normal
    end
    
    local hl = zombie:FindFirstChild("Highlight")
    
    if systemConfig.zombieHighlighter.HighlightEffect then
        if hl then
            hl.FillColor = color
            hl.OutlineColor = color
            hl.Enabled = true
        else
            hl = Instance.new("Highlight")
            hl.Name = "Highlight"
            hl.Adornee = zombie
            hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            hl.FillColor = color
            hl.FillTransparency = 0.5
            hl.OutlineColor = color
            hl.OutlineTransparency = 0
            hl.Parent = zombie
        end
    else
        if hl then
            hl.Enabled = false
        end
    end
end

local function initializeZombies()
    for _, z in ipairs(camera:GetChildren()) do
        if z:IsA("Model") and z.Name == "m_Zombie" then
            processZombie(z)
        end
    end
end

local function removeAllHighlights()
    for _, z in ipairs(camera:GetChildren()) do
        if z:IsA("Model") and z.Name == "m_Zombie" then
            local hl = z:FindFirstChild("Highlight")
            if hl then hl:Destroy() end
        end
    end
end

local function toggleHighlightEffects(enabled)
    for _, z in ipairs(camera:GetChildren()) do
        if z:IsA("Model") and z.Name == "m_Zombie" then
            local hl = z:FindFirstChild("Highlight")
            if hl then
                hl.Enabled = enabled
            end
        end
    end
end

-- ===== 速度控制功能 =====
local function applySpeedToPlayer()
    if not LocalPlayer then
        print("❌ 找不到本地玩家")
        return false
    end
    
    local character = LocalPlayer.Character
    if not character then
        local connection
        connection = LocalPlayer.CharacterAdded:Connect(function(newChar)
            task.wait(0.5)
            local humanoid = newChar:FindFirstChild("Humanoid")
            if humanoid then
                humanoid.WalkSpeed = systemConfig.speedSystem.customSpeed
            end
            connection:Disconnect()
        end)
        return false
    end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then
        print("❌ 找不到Humanoid")
        return false
    end
    
    humanoid.WalkSpeed = systemConfig.speedSystem.customSpeed
    return true
end

local function startSpeedLoop()
    if systemConfig.speedSystem.speedLoop then
        systemConfig.speedSystem.speedLoop:Disconnect()
        systemConfig.speedSystem.speedLoop = nil
    end
    
    systemConfig.speedSystem.speedLoop = game:GetService("RunService").Heartbeat:Connect(function()
        if not systemConfig.speedSystem.freezeSpeed then return end
        applySpeedToPlayer()
    end)
end

        systemConfig.speedSystem.freezeSpeed = value
        
        if value then
            applySpeedToPlayer()
            startSpeedLoop()
            Rayfield:Notify({
                Title = "速度锁定",
                Content = "速度已锁定: " .. systemConfig.speedSystem.customSpeed,
                Duration = 2
            })
        else
            if systemConfig.speedSystem.speedLoop then
                systemConfig.speedSystem.speedLoop:Disconnect()
                systemConfig.speedSystem.speedLoop = nil
            end
            Rayfield:Notify({
                Title = "速度锁定",
                Content = "速度锁定已关闭",
                Duration = 2
            })
        end
    end
})
